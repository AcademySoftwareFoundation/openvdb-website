450 template<typename Type>
451 inline bool
452 isRelOrApproxEqual(const Type& a, const Type& b, const Type& absTol, const Type& relTol)
453 {
454  // First check to see if we are inside the absolute tolerance
455  // Necessary for numbers close to 0
456  if (!(Abs(a - b) > absTol)) return true;
457 
458  // Next check to see if we are inside the relative tolerance
459  // to handle large numbers that aren't within the abs tolerance
460  // but could be the closest floating point representation
461  double relError;
462  if (Abs(b) > Abs(a)) {
463  relError = Abs((a - b) / b);
464  } else {
465  relError = Abs((a - b) / a);
466  }
467  return (relError <= relTol);
468 }

497 // aUnitsInLastPlace is the allowed difference between the least significant digits
498 // of the numbers' floating point representation
499 // Please read the reference paper before trying to use isUlpsEqual
500 // http://www.cygnus-software.com/papers/comparingfloats/comparingfloats.htm
501 inline bool
502 isUlpsEqual(const double aLeft, const double aRight, const int64_t aUnitsInLastPlace)
503 {
504  int64_t longLeft = doubleToInt64(aLeft);
505  // Because of 2's complement, must restore lexicographical order
506  if (longLeft < 0) {
507  longLeft = INT64_C(0x8000000000000000) - longLeft;
508  }
509 
510  int64_t longRight = doubleToInt64(aRight);
511  // Because of 2's complement, must restore lexicographical order
512  if (longRight < 0) {
513  longRight = INT64_C(0x8000000000000000) - longRight;
514  }
515 
516  int64_t difference = Abs(longLeft - longRight);
517  return (difference <= aUnitsInLastPlace);
518 }

522 {
