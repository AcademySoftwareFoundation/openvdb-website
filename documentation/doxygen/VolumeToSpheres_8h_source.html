432 void
433 ClosestPointDist<Index32LeafT>::evalNode(size_t pointIndex, size_t nodeIndex) const
434 {
435  if (nodeIndex >= mLeafRanges.size()) return;
436 
437  const Vec3R& pos = mInstancePoints[pointIndex];
438  float minDist = mInstanceDistances[pointIndex];
439  size_t minDistIdx = 0;
440  Vec3R center;
441  bool updatedDist = false;
442 
443  for (size_t i = mLeafRanges[nodeIndex].first, n = 0;
444  i < mLeafRanges[nodeIndex].second; ++i, ++n)
445  {
446  float& distToLeaf = const_cast<float&>(mLeafDistances[n]);
447 
448  center[0] = mLeafBoundingSpheres[i][0];
449  center[1] = mLeafBoundingSpheres[i][1];
450  center[2] = mLeafBoundingSpheres[i][2];
451  const auto radiusSqr = mLeafBoundingSpheres[i][3];
452 
453  distToLeaf = float(std::max(0.0, (pos - center).lengthSqr() - radiusSqr));
454 
455  if (distToLeaf < minDist) {
456  minDist = distToLeaf;
457  minDistIdx = i;
458  updatedDist = true;
459  }
460  }
461 
462  if (!updatedDist) return;
463 
464  evalLeaf(pointIndex, *mLeafNodes[minDistIdx]);
465 
466  for (size_t i = mLeafRanges[nodeIndex].first, n = 0;
467  i < mLeafRanges[nodeIndex].second; ++i, ++n)
468  {
469  if (mLeafDistances[n] < mInstanceDistances[pointIndex] && i != minDistIdx) {
470  evalLeaf(pointIndex, *mLeafNodes[i]);
471  }
472  }

629 template<typename GridT, typename InterrupterT>
630 inline void
631 fillWithSpheres(
632  const GridT& grid,
633  std::vector<openvdb::Vec4s>& spheres,
634  const Vec2i& sphereCount,
635  bool overlapping,
636  float minRadius,
637  float maxRadius,
638  float isovalue,
639  int instanceCount,
640  InterrupterT* interrupter)
641 {
642  spheres.clear();
643 
644  if (grid.empty()) return;
645 
646  const int
647  minSphereCount = sphereCount[0],
648  maxSphereCount = sphereCount[1];
649  if ((minSphereCount > maxSphereCount) || (maxSphereCount < 1)) {
650  OPENVDB_LOG_WARN("fillWithSpheres: minimum sphere count ("
651  << minSphereCount << ") exceeds maximum count (" << maxSphereCount << ")");
652  return;
653  }
654  spheres.reserve(maxSphereCount);
655 
656  auto gridPtr = grid.copy(); // shallow copy
657 
658  if (gridPtr->getGridClass() == GRID_LEVEL_SET) {
659  // Clamp the isovalue to the level set's background value minus epsilon.
660  // (In a valid narrow-band level set, all voxels, including background voxels,
661  // have values less than or equal to the background value, so an isovalue
662  // greater than or equal to the background value would produce a mask with
663  // effectively infinite extent.)
664  isovalue = std::min(isovalue,
665  static_cast<float>(gridPtr->background() - math::Tolerance<float>::value()));
666  } else if (gridPtr->getGridClass() == GRID_FOG_VOLUME) {
667  // Clamp the isovalue of a fog volume between epsilon and one,
668  // again to avoid a mask with infinite extent. (Recall that
669  // fog volume voxel values vary from zero outside to one inside.)
670  isovalue = math::Clamp(isovalue, math::Tolerance<float>::value(), 1.f);
671  }
672 
673  // ClosestSurfacePoint is inaccurate for small grids.
674  // Resample the input grid if it is too small.
675  auto numVoxels = gridPtr->activeVoxelCount();
676  if (numVoxels < 10000) {
677  const auto scale = 1.0 / math::Cbrt(2.0 * 10000.0 / double(numVoxels));
678  auto scaledXform = gridPtr->transform().copy();
679  scaledXform->preScale(scale);
680 
681  auto newGridPtr = levelSetRebuild(*gridPtr, isovalue,
682  LEVEL_SET_HALF_WIDTH, LEVEL_SET_HALF_WIDTH, scaledXform.get(), interrupter);
683 
684  const auto newNumVoxels = newGridPtr->activeVoxelCount();
685  if (newNumVoxels > numVoxels) {
686  OPENVDB_LOG_DEBUG_RUNTIME("fillWithSpheres: resampled input grid from "
687  << numVoxels << " voxel" << (numVoxels == 1 ? "" : "s")
688  << " to " << newNumVoxels << " voxel" << (newNumVoxels == 1 ? "" : "s"));
689  gridPtr = newGridPtr;
690  numVoxels = newNumVoxels;
691  }
692  }
693 
694  const bool addNarrowBandPoints = (numVoxels < 10000);
695  int instances = std::max(instanceCount, maxSphereCount);
696 
697  using TreeT = typename GridT::TreeType;
698  using BoolTreeT = typename TreeT::template ValueConverter<bool>::Type;
699  using Int16TreeT = typename TreeT::template ValueConverter<Int16>::Type;
700 
701  using RandGen = std::mersenne_twister_engine<uint32_t, 32, 351, 175, 19,
702  0xccab8ee7, 11, 0xffffffff, 7, 0x31b6ab00, 15, 0xffe50000, 17, 1812433253>; // mt11213b
703  RandGen mtRand(/*seed=*/0);
704 
705  const TreeT& tree = gridPtr->tree();
706  math::Transform transform = gridPtr->transform();
707 
708  std::vector<Vec3R> instancePoints;
709  {
710  // Compute a mask of the voxels enclosed by the isosurface.
711  typename Grid<BoolTreeT>::Ptr interiorMaskPtr;
712  if (gridPtr->getGridClass() == GRID_LEVEL_SET) {
713  interiorMaskPtr = sdfInteriorMask(*gridPtr, isovalue);
714  } else {
715  // For non-level-set grids, the interior mask comprises the active voxels.
716  interiorMaskPtr = typename Grid<BoolTreeT>::Ptr(Grid<BoolTreeT>::create(false));
717  interiorMaskPtr->setTransform(transform.copy());
718  interiorMaskPtr->tree().topologyUnion(tree);
719  }
720 
721  if (interrupter && interrupter->wasInterrupted()) return;
722 
723  // If the interior mask is small and eroding it results in an empty grid,
724  // use the uneroded mask instead. (But if the minimum sphere count is zero,
725  // then eroding away the mask is acceptable.)
726  if (!addNarrowBandPoints || (minSphereCount <= 0)) {
727  erodeVoxels(interiorMaskPtr->tree(), 1);
728  } else {
729  auto& maskTree = interiorMaskPtr->tree();
730  auto copyOfTree = StaticPtrCast<BoolTreeT>(maskTree.copy());
731  erodeVoxels(maskTree, 1);
732  if (maskTree.empty()) { interiorMaskPtr->setTree(copyOfTree); }
733  }
734 
735  // Scatter candidate sphere centroids (instancePoints)
736  instancePoints.reserve(instances);
737  v2s_internal::PointAccessor ptnAcc(instancePoints);
738 
739  const auto scatterCount = Index64(addNarrowBandPoints ? (instances / 2) : instances);
740 
741  UniformPointScatter<v2s_internal::PointAccessor, RandGen, InterrupterT> scatter(
742  ptnAcc, scatterCount, mtRand, 1.0, interrupter);
743  scatter(*interiorMaskPtr);
744  }
745 
746  if (interrupter && interrupter->wasInterrupted()) return;
747 
748  auto csp = ClosestSurfacePoint<GridT>::create(*gridPtr, isovalue, interrupter);
749  if (!csp) return;
750 
751  // Add extra instance points in the interior narrow band.
752  if (instancePoints.size() < size_t(instances)) {
753  const Int16TreeT& signTree = csp->signTree();
754  for (auto leafIt = signTree.cbeginLeaf(); leafIt; ++leafIt) {
755  for (auto it = leafIt->cbeginValueOn(); it; ++it) {
756  const int flags = int(it.getValue());
757  if (!(volume_to_mesh_internal::EDGES & flags)
758  && (volume_to_mesh_internal::INSIDE & flags))
759  {
760  instancePoints.push_back(transform.indexToWorld(it.getCoord()));
761  }
762  if (instancePoints.size() == size_t(instances)) break;
763  }
764  if (instancePoints.size() == size_t(instances)) break;
765  }
766  }
767 
768  if (interrupter && interrupter->wasInterrupted()) return;
769 
770  // Assign a radius to each candidate sphere. The radius is the world-space
771  // distance from the sphere's center to the closest surface point.
772  std::vector<float> instanceRadius;
773  if (!csp->search(instancePoints, instanceRadius)) return;
774 
775  float largestRadius = 0.0;
776  int largestRadiusIdx = 0;
777  for (size_t n = 0, N = instancePoints.size(); n < N; ++n) {
778  if (instanceRadius[n] > largestRadius) {
779  largestRadius = instanceRadius[n];
780  largestRadiusIdx = int(n);
781  }
782  }
783 
784  std::vector<unsigned char> instanceMask(instancePoints.size(), 0);
785 
786  minRadius = float(minRadius * transform.voxelSize()[0]);
787  maxRadius = float(maxRadius * transform.voxelSize()[0]);
788 
789  for (size_t s = 0, S = std::min(size_t(maxSphereCount), instancePoints.size()); s < S; ++s) {
790 
791  if (interrupter && interrupter->wasInterrupted()) return;
792 
793  largestRadius = std::min(maxRadius, largestRadius);
794 
795  if ((int(s) >= minSphereCount) && (largestRadius < minRadius)) break;
796 
797  const Vec4s sphere(
798  float(instancePoints[largestRadiusIdx].x()),
799  float(instancePoints[largestRadiusIdx].y()),
800  float(instancePoints[largestRadiusIdx].z()),
801  largestRadius);
802 
803  spheres.push_back(sphere);
804  instanceMask[largestRadiusIdx] = 1;
805 
806  v2s_internal::UpdatePoints op(
807  sphere, instancePoints, instanceRadius, instanceMask, overlapping);
808  op.run();
809 
810  largestRadius = op.radius();
811  largestRadiusIdx = op.index();
812  }
813 } // fillWithSpheres

831 template<typename InterrupterT>
832 inline bool
833 ClosestSurfacePoint<GridT>::initialize(
834  const GridT& grid, float isovalue, InterrupterT* interrupter)
835 {
836  using Index32LeafManagerT = tree::LeafManager<Index32TreeT>;
837  using ValueT = typename GridT::ValueType;
838 
839  const TreeT& tree = grid.tree();
840  const math::Transform& transform = grid.transform();
841 
842  { // Extract surface point cloud
843 
844  BoolTreeT mask(false);
845  volume_to_mesh_internal::identifySurfaceIntersectingVoxels(mask, tree, ValueT(isovalue));
846 
847  mSignTreePt.reset(new Int16TreeT(0));
848  mIdxTreePt.reset(new Index32TreeT(std::numeric_limits<Index32>::max()));
849 
850 
851  volume_to_mesh_internal::computeAuxiliaryData(
852  *mSignTreePt, *mIdxTreePt, mask, tree, ValueT(isovalue));
853 
854  if (interrupter && interrupter->wasInterrupted()) return false;
855 
856  // count unique points
857 
858  using Int16LeafNodeType = typename Int16TreeT::LeafNodeType;
859  using Index32LeafNodeType = typename Index32TreeT::LeafNodeType;
860 
861  std::vector<Int16LeafNodeType*> signFlagsLeafNodes;
862  mSignTreePt->getNodes(signFlagsLeafNodes);
863 
864  const tbb::blocked_range<size_t> auxiliaryLeafNodeRange(0, signFlagsLeafNodes.size());
865 
866  std::unique_ptr<Index32[]> leafNodeOffsets(new Index32[signFlagsLeafNodes.size()]);
867 
868  tbb::parallel_for(auxiliaryLeafNodeRange,
869  volume_to_mesh_internal::LeafNodePointCount<Int16LeafNodeType::LOG2DIM>
870  (signFlagsLeafNodes, leafNodeOffsets));
871 
872  {
873  Index32 pointCount = 0;
874  for (size_t n = 0, N = signFlagsLeafNodes.size(); n < N; ++n) {
875  const Index32 tmp = leafNodeOffsets[n];
876  leafNodeOffsets[n] = pointCount;
877  pointCount += tmp;
878  }
879 
880  mPointListSize = size_t(pointCount);
881  mSurfacePointList.reset(new Vec3s[mPointListSize]);
882  }
883 
884 
885  std::vector<Index32LeafNodeType*> pointIndexLeafNodes;
886  mIdxTreePt->getNodes(pointIndexLeafNodes);
887 
888  tbb::parallel_for(auxiliaryLeafNodeRange, volume_to_mesh_internal::ComputePoints<TreeT>(
889  mSurfacePointList.get(), tree, pointIndexLeafNodes,
890  signFlagsLeafNodes, leafNodeOffsets, transform, ValueT(isovalue)));
891  }
892 
893  if (interrupter && interrupter->wasInterrupted()) return false;
894 
895  Index32LeafManagerT idxLeafs(*mIdxTreePt);
896 
897  using Index32RootNodeT = typename Index32TreeT::RootNodeType;
898  using Index32NodeChainT = typename Index32RootNodeT::NodeChainType;
899  static_assert(Index32NodeChainT::Size > 1,
900  "expected tree depth greater than one");
901  using Index32InternalNodeT = typename Index32NodeChainT::template Get<1>;
902 
903  typename Index32TreeT::NodeCIter nIt = mIdxTreePt->cbeginNode();
904  nIt.setMinDepth(Index32TreeT::NodeCIter::LEAF_DEPTH - 1);
905  nIt.setMaxDepth(Index32TreeT::NodeCIter::LEAF_DEPTH - 1);
906 
907  std::vector<const Index32InternalNodeT*> internalNodes;
908 
909  const Index32InternalNodeT* node = nullptr;
910  for (; nIt; ++nIt) {
911  nIt.getNode(node);
912  if (node) internalNodes.push_back(node);
913  }
914 
915  std::vector<IndexRange>().swap(mLeafRanges);
916  mLeafRanges.resize(internalNodes.size());
917 
918  std::vector<const Index32LeafT*>().swap(mLeafNodes);
919  mLeafNodes.reserve(idxLeafs.leafCount());
920 
921  typename Index32InternalNodeT::ChildOnCIter leafIt;
922  mMaxNodeLeafs = 0;
923  for (size_t n = 0, N = internalNodes.size(); n < N; ++n) {
924 
925  mLeafRanges[n].first = mLeafNodes.size();
926 
927  size_t leafCount = 0;
928  for (leafIt = internalNodes[n]->cbeginChildOn(); leafIt; ++leafIt) {
929  mLeafNodes.push_back(&(*leafIt));
930  ++leafCount;
931  }
932 
933  mMaxNodeLeafs = std::max(leafCount, mMaxNodeLeafs);
934 
935  mLeafRanges[n].second = mLeafNodes.size();
936  }
937 
938  std::vector<Vec4R>().swap(mLeafBoundingSpheres);
939  mLeafBoundingSpheres.resize(mLeafNodes.size());
940 
941  v2s_internal::LeafOp<Index32LeafT> leafBS(
942  mLeafBoundingSpheres, mLeafNodes, transform, mSurfacePointList);
943  leafBS.run();
944 
945 
946  std::vector<Vec4R>().swap(mNodeBoundingSpheres);
947  mNodeBoundingSpheres.resize(internalNodes.size());
948 
949  v2s_internal::NodeOp nodeBS(mNodeBoundingSpheres, mLeafRanges, mLeafBoundingSpheres);
950  nodeBS.run();
951  return true;
952 } // ClosestSurfacePoint::initialize
