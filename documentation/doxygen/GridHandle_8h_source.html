360 template<typename BufferT>
361 void GridHandle<BufferT>::read(std::istream& is, const BufferT& pool)
362 {
363  GridData data;
364  is.read((char*)&data, sizeof(GridData));
365  if (data.isValid()) {
366  uint64_t size = data.mGridSize, sum = 0u;
367  while(data.mGridIndex + 1u < data.mGridCount) {// loop over remaining raw grids in stream
368  is.seekg(data.mGridSize - sizeof(GridData), std::ios::cur);// skip grid
369  is.read((char*)&data, sizeof(GridData));
370  sum += data.mGridSize;
371  }
372  auto buffer = BufferT::create(size + sum, &pool);
373  is.seekg(-int64_t(sum + sizeof(GridData)), std::ios::cur);// rewind to start
374  is.read((char*)(buffer.data()), buffer.size());

376  } else {

430 /// @brief Split all grids in a single GridHandle into a vector of multiple GridHandles each with a single grid
431 /// @tparam BufferT Type of the input and output grid buffers

435 template<typename BufferT, template <class, class...> class VectorT = std::vector>
436 inline VectorT<GridHandle<BufferT>>
437 splitGrids(const GridHandle<BufferT> &handle, const BufferT* other = nullptr)
438 {
439  using HandleT = GridHandle<BufferT>;
440  const void *ptr = handle.data();
441  if (ptr == nullptr) return VectorT<HandleT>();
442  VectorT<HandleT> handles(handle.gridCount());
443  for (auto &h : handles) {
444  const GridData *src = reinterpret_cast<const GridData*>(ptr);
445  NANOVDB_ASSERT(src->isValid());
446  auto buffer = BufferT::create(src->mGridSize, other);
447  GridData *dst = reinterpret_cast<GridData*>(buffer.data());
448  std::memcpy(dst, src, src->mGridSize);
449  tools::updateGridCount(dst, 0u, 1u);
450  h = HandleT(std::move(buffer));
451  ptr = util::PtrAdd(ptr, src->mGridSize);
452  }
453  return std::move(handles);
454 }// splitGrids

456 /// @brief Combines (or merges) multiple GridHandles into a single GridHandle containing all grids
457 /// @tparam BufferT Type of the input and output grid buffers
458 /// @param handles Vector of GridHandles to be combined
459 /// @param pool optional pool used for allocation of output GridHandle
460 /// @return single GridHandle containing all input grids
461 template<typename BufferT, template <class, class...> class VectorT>
462 inline GridHandle<BufferT>
463 mergeGrids(const VectorT<GridHandle<BufferT>> &handles, const BufferT* pool = nullptr)
464 {
465  uint64_t size = 0u;
466  uint32_t counter = 0u, gridCount = 0u;
467  for (auto &h : handles) {
468  gridCount += h.gridCount();
469  for (uint32_t n=0; n<h.gridCount(); ++n) size += h.gridSize(n);
470  }
471  auto buffer = BufferT::create(size, pool);
472  void *dst = buffer.data();
473  for (auto &h : handles) {
474  const void *src = h.data();
475  for (uint32_t n=0; n<h.gridCount(); ++n) {
476  std::memcpy(dst, src, h.gridSize(n));
477  GridData *data = reinterpret_cast<GridData*>(dst);
478  NANOVDB_ASSERT(data->isValid());
479  tools::updateGridCount(data, counter++, gridCount);
480  dst = util::PtrAdd(dst, data->mGridSize);
481  src = util::PtrAdd(src, data->mGridSize);
482  }
483  }
484  return GridHandle<BufferT>(std::move(buffer));
485 }// mergeGrids
