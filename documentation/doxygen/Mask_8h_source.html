114 template<typename GridType>
115 inline typename std::enable_if<!(mask_internal::Traits<GridType>::isBool),
116  typename mask_internal::Traits<GridType>::BoolGridPtrType>::type
117 doInteriorMask(const GridType& grid, const double isovalue)
118 {
119  using MaskGridT = typename mask_internal::Traits<GridType>::BoolGridType;
120 
121  // If the input grid is a level set, return a mask of its interior.
122  if (auto maskGridPtr = doLevelSetInteriorMask(grid, isovalue)) {
123  return maskGridPtr;
124  }
125 
126  // For any other grid type, return a mask of its active voxels.
127  auto maskGridPtr = MaskGridT::create(/*background=*/false);
128  maskGridPtr->setTransform(grid.transform().copy());
129  maskGridPtr->topologyUnion(grid);
130  return maskGridPtr;
131 }
