123/// @brief Namespace to hold functionality for registering info text callbacks. Whenever
124/// getNodeSpecificInfoText() is called, the default info text is added to MMB output unless
125/// a valid callback has been registered for the grid type.
126///
127/// @details Use node_info_text::registerGridSpecificInfoText<> to register a grid type to
128/// a function pointer which matches the ApplyGridSpecificInfoText signature.
129///
130/// void floatGridText(std::ostream&, const openvdb::GridBase&);
131///
132/// node_info_text::registerGridSpecificInfoText<openvdb::FloatGrid>(&floatGridText);
133///
134namespace node_info_text
135{
136 // The function pointer signature expected when registering an grid type text
137 // callback. The grid is passed untyped but is guaranteed to match the registered
138 // type.
139 using ApplyGridSpecificInfoText = void (*)(std::ostream&, const openvdb::GridBase&);
140
141 /// @brief Register an info text callback to a specific grid type.
142 /// @note Does not add the callback if the grid type already has a registered callback.
143 /// @param gridType the grid type as a unique string (see templated
144 /// registerGridSpecificInfoText<>)
145 /// @param callback a pointer to the callback function to execute
146 void registerGridSpecificInfoText(const std::string& gridType,
147 ApplyGridSpecificInfoText callback);
148
149 /// @brief Register an info text callback to a templated grid type.
150 /// @note Does not add the callback if the grid type already has a registered callback.
151 /// @param callback a pointer to the callback function to execute
152 template<typename GridType>
153 inline void registerGridSpecificInfoText(ApplyGridSpecificInfoText callback)
154 {
155 registerGridSpecificInfoText(GridType::gridType(), callback);
156 }
157
158} // namespace node_info_text
